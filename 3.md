<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Paragraph Context Example</title>
</head>

<body>

    <!-- Paragraph 1: Direct child of body (comes before the div) -->
    <p>
        1st paragraph – Direct child of

        <body>.
            Lorem ipsum dolor sit amet.
    </p>

    <!-- A container div -->
    <div class="div">

        <!-- Paragraph 2: Direct child of the div (first <p> inside div) -->
        <p>
            2nd paragraph – First direct child of the div.
            Lorem ipsum dolor sit amet consectetur adipisicing elit. Ut eum id reprehenderit
            odit vel tempore delectus cumque quia repellat a!
        </p>

        <!-- Span is a direct child of the div -->
        <span>

            <!-- Paragraph 3: Direct child of span, and descendant of the div -->
            <p>
                3rd paragraph – First child of span and a descendant of the div.
                Lorem ipsum dolor sit amet consectetur adipisicing elit. Minima, magnam.
            </p>

        </span>

        <!-- Paragraph 4: Direct child of the div (second <p> inside div) -->
        <p>
            4th paragraph – Second direct child of the div.
            Lorem ipsum dolor sit amet, consectetur adipisicing elit. Fugiat numquam,
            sit excepturi libero nulla nisi.
        </p>

    </div>

    <!-- Paragraph 5: Immediately after the div (adjacent sibling of div) -->
    <p>
        5th paragraph – Immediately after the div.
        This paragraph is the adjacent sibling of the div.
        Lorem ipsum dolor sit amet consectetur adipisicing elit.
    </p>

    <!-- Paragraph 6: After the div but NOT immediately (general sibling of div) -->
    <p>
        6th paragraph – After the div but not immediately.
        This paragraph is a general sibling of the div.
        Lorem ipsum dolor sit amet, consectetur adipisicing elit. Perspiciatis, ullam?
    </p>

</body>

</html>
